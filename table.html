1. ketik ssh-keygen -t rsa "untuk generate ke ssh key di github (1x)"
2. ketik cat ~/.ssh/id_rsa.pub "copy ssh key ke github ssh key (1x)"
3. ketik mkdir (nama folder) "cara membuat folder"
4. ketik cd project "masuk ke folder"
5. buat file .html di sublime
6. ketik git init "untuk memakai git"
7. ketik git add index,html "untuk mengadd file html ke github"
8. ketik git status "untuk melihat status commit apa blom"
9. ketik git commit "untuk memberikan keterangan file yang di push"
10. ketik git config --global user.email "masukan email anda" (1x)
11. KETIK git config --global user.nama "masukan nama anda" (1x)
12. ketik git remote add origin "(masukan ssh url project yg di github) "
13. ketik git remote -v "untuk melihat akses project"
14. ketik git push origin master "untuk mengupload ke github"
15. ketik git pull origin master "untuk mendowload jeng merge file-file"
16. ketik git checkout -b (nama branch) "untuk menambahkan branch"
17. ketik git checkout (nama branch) "untuk pindah ke branch yang lain"
